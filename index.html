main.go 
import (
    optic "github.com/observability/optic-client-go"
)

func main() {
    // Initialize with your API key
    client := optic.NewClient(optic.Config{
        Endpoint: "https://ingest.optikk.io/otlp",
        APIKey:   os.Getenv("OPTIKK_API_KEY"),
        Service:  "payment-service",
    })
    defer client.Shutdown(ctx)

    // Trace an HTTP handler automatically
    http.Handle("/checkout", client.Middleware(handler))

    // Emit a custom metric
    client.Metrics.Counter("orders.created",
        1, map[string]string{
            "region": "us-east-1",
        })
}
from optic_client import OpticClient, Config

# Initialize the client
client = OpticClient(Config(
    endpoint="https://ingest.optikk.io/otlp",
    api_key=os.environ["OPTIKK_API_KEY"],
    service="ml-inference",
))

# Decorate any function for automatic tracing
@client.trace("predict")
def predict(features):
    result = model.infer(features)
    return result

# Track an AI request
with client.ai_span(model="gpt-4o") as span:
    resp = openai.chat.completions.create(...)
    span.set_tokens(resp.usage.total_tokens)
// Add to pom.xml or build.gradle
OpticClient client = OpticClient.builder()
    .endpoint("https://ingest.optikk.io/otlp")
    .apiKey(System.getenv("OPTIKK_API_KEY"))
    .serviceName("order-service")
    .build();

// Spring Boot auto-instrumentation
@Bean
public OpticTracingFilter tracingFilter() {
    return new OpticTracingFilter(client);
}

// Manual span
try (Span span = client.startSpan("process-order")) {
    span.setAttribute("order.id", orderId);
    processOrder(orderId);
}
